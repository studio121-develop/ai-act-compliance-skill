# .NET — Compliance for the Other Enterprise Framework

If Spring Boot is the enterprise Java answer to AI Act compliance, ASP.NET Core is the
enterprise C# answer. Same patterns, different curly braces, arguably better tooling.
Here's everything you need: middleware, entity base class, Razor component, action filter,
and structured logging. Your compliance auditor won't know the difference.

## ASP.NET Core Middleware

```csharp
// AIComplianceMiddleware.cs
namespace YourApp.Compliance;

public class AIComplianceMiddleware
{
    private readonly RequestDelegate _next;
    private readonly AIComplianceOptions _options;
    private static readonly string[] AiRoutePrefixes = ["/api/ai/", "/api/chat/", "/api/generate/"];

    public AIComplianceMiddleware(RequestDelegate next, AIComplianceOptions options)
    {
        _next = next;
        _options = options;
    }

    public async Task InvokeAsync(HttpContext context)
    {
        var path = context.Request.Path.Value ?? "";
        var isAiRoute = AiRoutePrefixes.Any(prefix => path.StartsWith(prefix, StringComparison.OrdinalIgnoreCase));

        if (isAiRoute)
        {
            context.Response.Headers.Append("X-AI-Generated", "true");
            context.Response.Headers.Append("X-AI-Provider", _options.Provider);
            context.Response.Headers.Append("X-AI-System", _options.SystemName);
            context.Response.Headers.Append("X-AI-Act-Disclosure", "Article 50(1)");
            context.Response.Headers.Append("X-AI-Timestamp", DateTime.UtcNow.ToString("o"));
        }

        await _next(context);
    }
}

public record AIComplianceOptions
{
    public string Provider { get; init; } = "unknown";
    public string SystemName { get; init; } = "unknown";
    public string Model { get; init; } = "unknown";
    public bool AuditEnabled { get; init; } = true;
    public int AuditRetentionDays { get; init; } = 365;
}

public static class AIComplianceExtensions
{
    public static IApplicationBuilder UseAICompliance(
        this IApplicationBuilder app, AIComplianceOptions options)
    {
        return app.UseMiddleware<AIComplianceMiddleware>(options);
    }
}
```

## Entity Framework Core Base Class

```csharp
// AIContentEntity.cs
namespace YourApp.Compliance;

using System.ComponentModel.DataAnnotations;
using System.ComponentModel.DataAnnotations.Schema;

public abstract class AIContentEntity
{
    [Required]
    public bool AiGenerated { get; init; }

    [MaxLength(100)]
    public string AiModel { get; init; } = "";

    [MaxLength(200)]
    public string AiSystem { get; init; } = "";

    public DateTime? AiTimestamp { get; init; }

    [Required]
    public bool AiHumanEdited { get; init; }

    [NotMapped]
    public string AiDisclosureText
    {
        get
        {
            if (!AiGenerated) return "";
            var edited = AiHumanEdited ? " (human-edited)" : "";
            return $"Generated by AI system: {AiSystem}, model: {AiModel}{edited}";
        }
    }
}
```

## Razor Component for AI Disclosure

```razor
@* Components/AIDisclosure.razor *@

<div role="status"
     aria-label="AI disclosure notice"
     data-ai-disclosure="true"
     class="ai-disclosure-banner">
    <p><strong>AI-Powered Interaction</strong></p>
    <p>
        This feature uses <strong>@SystemName</strong> by @ProviderName.
        Content generated by this system is clearly marked.

    </p>
    @if (ShowAcknowledge)
    {
        @if (!_acknowledged)
        {
            <button @onclick="Acknowledge"
                    class="btn btn-warning mt-2">
                I understand this is AI-generated
            </button>
        }
        else
        {
            <span class="badge bg-secondary mt-2">Disclosure acknowledged</span>
        }
    }
</div>

@code {
    [Parameter] public string SystemName { get; set; } = "AI System";
    [Parameter] public string ProviderName { get; set; } = "AI Provider";
    [Parameter] public bool ShowAcknowledge { get; set; } = false;
    [Parameter] public EventCallback OnAcknowledged { get; set; }

    private bool _acknowledged;

    private async Task Acknowledge()
    {
        _acknowledged = true;
        await OnAcknowledged.InvokeAsync();
    }
}
```

## Action Filter for AI Metadata Injection

```csharp
// AIMetadataActionFilter.cs
namespace YourApp.Compliance;

using Microsoft.AspNetCore.Mvc;
using Microsoft.AspNetCore.Mvc.Filters;

public class AIMetadataActionFilter : IAsyncResultFilter
{
    private readonly AIComplianceOptions _options;

    public AIMetadataActionFilter(AIComplianceOptions options)
    {
        _options = options;
    }

    public async Task OnResultExecutionAsync(ResultExecutingContext context, ResultExecutionDelegate next)
    {
        if (context.Result is ObjectResult objectResult && objectResult.Value is not null)
        {
            var wrapped = new
            {
                data = objectResult.Value,
                compliance = new
                {
                    ai_generated = true,
                    ai_provider = _options.Provider,
                    ai_model = _options.Model,
                    ai_system = _options.SystemName,
                    timestamp = DateTime.UtcNow.ToString("o"),
                    disclosure = "Content generated by AI per EU AI Act Art. 50"
                }
            };
            objectResult.Value = wrapped;
        }

        await next();
    }
}
```

## ILogger-Based Compliance Audit Logging

```csharp
// ComplianceAuditLogger.cs
namespace YourApp.Compliance;

using Microsoft.Extensions.Logging;

public interface IComplianceAuditLogger
{
    void LogInteraction(string endpoint, string method, string? userId,
                        string requestSummary, string responseSummary,
                        AIComplianceOptions options);
}

public class ComplianceAuditLogger : IComplianceAuditLogger
{
    private readonly ILogger<ComplianceAuditLogger> _logger;

    public ComplianceAuditLogger(ILogger<ComplianceAuditLogger> logger)
    {
        _logger = logger;
    }

    public void LogInteraction(string endpoint, string method, string? userId,
                               string requestSummary, string responseSummary,
                               AIComplianceOptions options)
    {
        _logger.LogInformation(
            "AI interaction: Endpoint={Endpoint} Method={Method} UserId={UserId} " +
            "Provider={Provider} Model={Model} System={System} Timestamp={Timestamp}",
            endpoint, method, userId ?? "anonymous",
            options.Provider, options.Model, options.SystemName,
            DateTime.UtcNow.ToString("o")
        );
    }
}
```

## Full Controller Example

```csharp
// AIChatController.cs
namespace YourApp.Controllers;

using Microsoft.AspNetCore.Mvc;
using YourApp.Compliance;

[ApiController]
[Route("api/ai")]
[ServiceFilter(typeof(AIMetadataActionFilter))]
public class AIChatController : ControllerBase
{
    private readonly AIComplianceOptions _options;
    private readonly IComplianceAuditLogger _auditLogger;

    public AIChatController(AIComplianceOptions options, IComplianceAuditLogger auditLogger)
    {
        _options = options;
        _auditLogger = auditLogger;
    }

    public record ChatRequest(string Message, string? ConversationId);
    public record ChatResponse(string Content);

    [HttpPost("chat")]
    public IActionResult Chat([FromBody] ChatRequest request)
    {
        if (string.IsNullOrWhiteSpace(request.Message))
            return BadRequest(new { error = "message is required" });

        // Replace with your actual AI call
        var aiContent = $"AI response to: {request.Message}";

        _auditLogger.LogInteraction(
            endpoint: "/api/ai/chat",
            method: "POST",
            userId: User.Identity?.Name,
            requestSummary: request.Message[..Math.Min(request.Message.Length, 200)],
            responseSummary: aiContent[..Math.Min(aiContent.Length, 200)],
            options: _options
        );

        return Ok(new ChatResponse(aiContent));
    }
}
```

## Program.cs Registration

```csharp
// Program.cs (relevant excerpt)
var options = new AIComplianceOptions
{
    Provider = builder.Configuration["AI:Provider"] ?? "unknown",
    SystemName = builder.Configuration["AI:SystemName"] ?? "unknown",
    Model = builder.Configuration["AI:Model"] ?? "unknown",
};

builder.Services.AddSingleton(options);
builder.Services.AddScoped<AIMetadataActionFilter>();
builder.Services.AddSingleton<IComplianceAuditLogger, ComplianceAuditLogger>();

var app = builder.Build();
app.UseAICompliance(options);
```
