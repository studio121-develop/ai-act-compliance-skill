# Django — Old Reliable Meets New Regulation

Django has survived every web trend since 2005. It will survive the AI Act too.
Here's everything you need: middleware for headers, a model mixin so your AI content
carries its own metadata, a template tag that even your designer can use, and a
management command for the inevitable "prove we're compliant" conversation.

## Middleware: Compliance Headers

```python
# ai_compliance/middleware.py
from datetime import datetime, timezone
from django.conf import settings
from django.http import HttpRequest, HttpResponse

AI_ROUTE_PREFIXES = getattr(settings, "AI_ROUTE_PREFIXES", ("/api/ai/", "/api/chat/"))


class AIComplianceMiddleware:
    def __init__(self, get_response):
        self.get_response = get_response
        self.ai_provider = getattr(settings, "AI_PROVIDER", "unknown")
        self.ai_system = getattr(settings, "AI_SYSTEM_NAME", "unknown")

    def __call__(self, request: HttpRequest) -> HttpResponse:
        response = self.get_response(request)

        if any(request.path.startswith(prefix) for prefix in AI_ROUTE_PREFIXES):
            response["X-AI-Generated"] = "true"
            response["X-AI-Provider"] = self.ai_provider
            response["X-AI-System"] = self.ai_system
            response["X-AI-Act-Disclosure"] = "Article 50(1)"
            response["X-AI-Timestamp"] = datetime.now(timezone.utc).isoformat()

        return response
```

## Model Mixin: AI Content Metadata

```python
# ai_compliance/models.py
from django.db import models
from django.conf import settings


class AIContentMixin(models.Model):
    """Add to any model that may contain AI-generated content.

    Usage:
        class BlogPost(AIContentMixin, models.Model):
            title = models.CharField(max_length=200)
            body = models.TextField()
    """

    ai_generated = models.BooleanField(
        default=False,
        help_text="Whether this content was generated by an AI system.",
        db_index=True,
    )
    ai_model = models.CharField(
        max_length=100,
        blank=True,
        default="",
        help_text="The AI model used to generate this content (e.g. gpt-4o).",
    )
    ai_system = models.CharField(
        max_length=200,
        blank=True,
        default="",
        help_text="The AI system name per AI Act Art. 50.",
    )
    ai_timestamp = models.DateTimeField(
        null=True,
        blank=True,
        help_text="When the AI generated this content.",
    )
    ai_human_edited = models.BooleanField(
        default=False,
        help_text="Whether a human edited the AI-generated content after generation.",
    )

    class Meta:
        abstract = True

    @property
    def ai_disclosure_text(self) -> str:
        if not self.ai_generated:
            return ""
        edited = " (human-edited)" if self.ai_human_edited else ""
        return f"Generated by AI system: {self.ai_system}, model: {self.ai_model}{edited}"
```

## Template Tag: AI Disclosure Banner

```python
# ai_compliance/templatetags/ai_compliance_tags.py
from django import template
from django.utils.safestring import mark_safe
from django.conf import settings

register = template.Library()


@register.inclusion_tag("ai_compliance/disclosure_banner.html", takes_context=True)
def ai_disclosure(context, system_name=None, provider_name=None):
    return {
        "system_name": system_name or getattr(settings, "AI_SYSTEM_NAME", "AI System"),
        "provider_name": provider_name or getattr(settings, "AI_PROVIDER", "AI Provider"),
    }
```

## Management Command: Compliance Audit

```python
# ai_compliance/management/commands/ai_compliance_audit.py
from django.core.management.base import BaseCommand
from django.apps import apps


class Command(BaseCommand):
    help = "Audit all AI-generated content for missing metadata"

    def handle(self, *args, **options):
        issues = []
        total_ai_content = 0

        for model in apps.get_models():
            if not hasattr(model, "ai_generated"):
                continue

            ai_records = model.objects.filter(ai_generated=True)
            count = ai_records.count()
            total_ai_content += count

            missing_model = ai_records.filter(ai_model="").count()
            missing_system = ai_records.filter(ai_system="").count()
            missing_timestamp = ai_records.filter(ai_timestamp__isnull=True).count()

            label = model._meta.label
            if missing_model > 0:
                issues.append(f"{label}: {missing_model} records missing ai_model")
            if missing_system > 0:
                issues.append(f"{label}: {missing_system} records missing ai_system")
            if missing_timestamp > 0:
                issues.append(f"{label}: {missing_timestamp} records missing ai_timestamp")

        self.stdout.write(f"\nTotal AI-generated records: {total_ai_content}")
        if issues:
            self.stdout.write(self.style.WARNING(f"\nFound {len(issues)} issues:"))
            for issue in issues:
                self.stdout.write(self.style.WARNING(f"  - {issue}"))
        else:
            self.stdout.write(self.style.SUCCESS("\nAll AI content has complete metadata."))
```

## DRF Serializer Mixin

```python
# ai_compliance/serializers.py
from rest_framework import serializers


class AIMetadataSerializerMixin(serializers.Serializer):
    """Mix into any serializer whose model uses AIContentMixin."""

    ai_generated = serializers.BooleanField(read_only=True)
    ai_model = serializers.CharField(read_only=True)
    ai_system = serializers.CharField(read_only=True)
    ai_timestamp = serializers.DateTimeField(read_only=True)
    ai_human_edited = serializers.BooleanField(read_only=True)
    ai_disclosure_text = serializers.CharField(read_only=True, source="ai_disclosure_text")
```

## Signal-Based Compliance Logging

```python
# ai_compliance/signals.py
import logging
from django.db.models.signals import post_save
from django.dispatch import receiver
from django.apps import apps

logger = logging.getLogger("ai_compliance")


def register_ai_compliance_signals():
    """Call this in your AppConfig.ready() method."""
    for model in apps.get_models():
        if hasattr(model, "ai_generated"):
            post_save.connect(_log_ai_content_save, sender=model)


@receiver(post_save)
def _log_ai_content_save(sender, instance, created, **kwargs):
    if not hasattr(instance, "ai_generated") or not instance.ai_generated:
        return

    action = "created" if created else "updated"
    logger.info(
        "ai_content_%s" % action,
        extra={
            "model": sender._meta.label,
            "pk": instance.pk,
            "ai_model": instance.ai_model,
            "ai_system": instance.ai_system,
            "ai_human_edited": instance.ai_human_edited,
        },
    )
```
