# Next.js — Compliance in the App Router Era

You picked the React meta-framework that changes its routing model every 18 months.
The good news: App Router actually makes AI Act compliance *easier* — Server Components
for guaranteed server-side disclosure, middleware for headers, and Server Actions for
audit logging. Let's wire it all up.

## Middleware: Compliance Headers on Every AI Response

```typescript
// middleware.ts
import { NextResponse } from "next/server";
import type { NextRequest } from "next/server";

const AI_ROUTE_PREFIXES = ["/api/ai/", "/api/chat/", "/api/generate/"];

export function middleware(request: NextRequest) {
  const isAiRoute = AI_ROUTE_PREFIXES.some((prefix) =>
    request.nextUrl.pathname.startsWith(prefix)
  );

  const response = NextResponse.next();

  if (isAiRoute) {
    response.headers.set("X-AI-Generated", "true");
    response.headers.set(
      "X-AI-Provider",
      process.env.AI_PROVIDER ?? "unknown"
    );
    response.headers.set(
      "X-AI-System",
      process.env.AI_SYSTEM_NAME ?? "unknown"
    );
    response.headers.set("X-AI-Act-Disclosure", "Article 50(1)");
    response.headers.set("X-AI-Timestamp", new Date().toISOString());
  }

  return response;
}

export const config = {
  matcher: ["/api/ai/:path*", "/api/chat/:path*", "/api/generate/:path*"],
};
```

## Server Component: AI Disclosure Banner

This renders on the server — no hydration flicker, no "flash of non-disclosure."

```tsx
// components/ai-disclosure-banner.tsx
import { ReactNode } from "react";

interface AIDisclosureBannerProps {
  systemName: string;
  providerName: string;
  children?: ReactNode;
}

  systemName,
  providerName,
  children,

  return (
    <div
      role="status"

        generated by this system is clearly marked. Per EU AI Act Article 50,
        you have the right to know when you are interacting with an AI system.
      </p>
      {children}
    </div>
  );
}
```

## Client Component: Interactive Disclosure with Acknowledge

```tsx
"use client";

import { useState, useCallback } from "react";

interface AIDisclosureInteractiveProps {
  systemName: string;
  onAcknowledge?: () => void;
}

export function AIDisclosureInteractive({
  systemName,
  onAcknowledge,
}: AIDisclosureInteractiveProps) {
  const [acknowledged, setAcknowledged] = useState(false);

  const handleAcknowledge = useCallback(() => {
    setAcknowledged(true);
    onAcknowledge?.();
  }, [onAcknowledge]);

  if (acknowledged) {
    return (
      <div
        data-ai-disclosure="acknowledged"
        className="rounded bg-gray-100 px-3 py-1 text-xs text-gray-600"
      >
        AI system: {systemName} — Disclosure acknowledged
      </div>
    );
  }

  return (
    <div
      role="alertdialog"
      aria-label="AI disclosure acknowledgement required"
      data-ai-disclosure="pending"
      className="rounded-md border border-amber-300 bg-amber-50 p-4"
    >
      <p className="text-sm font-medium text-amber-900">
        You are about to interact with an AI system ({systemName}).
      </p>
      <p className="mt-1 text-xs text-amber-700">
        Responses are machine-generated and may contain errors. This disclosure
        is required by EU AI Act Article 50.
      </p>
      <button
        onClick={handleAcknowledge}
        className="mt-3 rounded bg-amber-600 px-4 py-1.5 text-sm text-white hover:bg-amber-700"
      >
        I understand this is AI-generated
      </button>
    </div>
  );
}
```

## API Route: Chat Endpoint with Full Compliance

```typescript
// app/api/chat/route.ts
import { NextRequest, NextResponse } from "next/server";
import { logAIInteraction } from "@/lib/compliance-logger";

interface ChatRequest {
  message: string;
  conversationId?: string;
}

interface ComplianceMetadata {
  ai_generated: true;
  ai_provider: string;
  ai_model: string;
  ai_system: string;
  timestamp: string;
  disclosure: string;
}

function buildComplianceMetadata(): ComplianceMetadata {
  return {
    ai_generated: true,
    ai_provider: process.env.AI_PROVIDER ?? "unknown",
    ai_model: process.env.AI_MODEL ?? "unknown",
    ai_system: process.env.AI_SYSTEM_NAME ?? "unknown",
    timestamp: new Date().toISOString(),
    disclosure: "Content generated by AI per EU AI Act Art. 50",
  };
}

export async function POST(request: NextRequest) {
  const body: ChatRequest = await request.json();

  if (!body.message || typeof body.message !== "string") {
    return NextResponse.json(
      { error: "message is required and must be a string" },
      { status: 400 }
    );
  }

  const metadata = buildComplianceMetadata();

  // Replace with your actual AI provider call
  const aiResponse = await fetch("https://api.your-ai-provider.com/chat", {
    method: "POST",
    headers: {
      Authorization: `Bearer ${process.env.AI_API_KEY}`,
      "Content-Type": "application/json",
    },
    body: JSON.stringify({ prompt: body.message }),
  });

  const aiData = await aiResponse.json();

  // Fire-and-forget audit log — don't block the response
  logAIInteraction({
    endpoint: "/api/chat",
    userMessage: body.message,
    aiResponse: aiData.content,
    metadata,
    conversationId: body.conversationId ?? null,
  }).catch(console.error);

  return NextResponse.json({
    content: aiData.content,
    compliance: metadata,
  });
}
```

## Server Action: Compliance Audit Logging

```typescript
// lib/compliance-logger.ts
"use server";

interface AIInteractionLog {
  endpoint: string;
  userMessage: string;
  aiResponse: string;
  metadata: {
    ai_generated: boolean;
    ai_provider: string;
    ai_model: string;
    ai_system: string;
    timestamp: string;
    disclosure: string;
  };
  conversationId: string | null;
}

export async function logAIInteraction(log: AIInteractionLog): Promise<void> {
  // Replace with your database/logging service of choice
  const record = {
    ...log,
    logged_at: new Date().toISOString(),
    retention_days: 365, // AI Act requires adequate record-keeping
  };

  // Example: write to your database
  await fetch(process.env.AUDIT_LOG_ENDPOINT ?? "http://localhost:3001/logs", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(record),
  });
}
```
