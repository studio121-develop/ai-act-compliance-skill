# Spring Boot — Enterprise Compliance, Enterprise Framework

If your company already has a Spring Boot monolith, congratulations: you now get to add
AI Act compliance to it. The good news is that Spring's interceptor/AOP/advice trinity
is genuinely useful here. The bad news is you'll write some XML. Just kidding — it's 2026,
we use application.yml now.

## application.yml Configuration

```yaml
# application.yml
ai:
  compliance:
    enabled: true
    provider: ${AI_PROVIDER:unknown}
    system-name: ${AI_SYSTEM_NAME:unknown}
    model: ${AI_MODEL:unknown}
    risk-level: LIMITED
    route-prefixes:
      - /api/ai/
      - /api/chat/
      - /api/generate/
    audit:
      enabled: true
      retention-days: 365
```

## HandlerInterceptor for Compliance Headers

```java
// AIComplianceInterceptor.java
package com.example.compliance;

import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import org.springframework.stereotype.Component;
import org.springframework.web.servlet.HandlerInterceptor;
import java.time.Instant;

@Component
public class AIComplianceInterceptor implements HandlerInterceptor {

    private final AIComplianceProperties props;

    public AIComplianceInterceptor(AIComplianceProperties props) {
        this.props = props;
    }

    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response,
                             Object handler) {
        if (!props.enabled()) return true;

        String path = request.getRequestURI();
        boolean isAiRoute = props.routePrefixes().stream().anyMatch(path::startsWith);

        if (isAiRoute) {
            response.setHeader("X-AI-Generated", "true");
            response.setHeader("X-AI-Provider", props.provider());
            response.setHeader("X-AI-System", props.systemName());
            response.setHeader("X-AI-Act-Disclosure", "Article 50(1)");
            response.setHeader("X-AI-Timestamp", Instant.now().toString());
        }

        return true;
    }
}
```

## JPA Entity with AI Metadata (@MappedSuperclass)

```java
// AIContentEntity.java
package com.example.compliance;

import jakarta.persistence.Column;
import jakarta.persistence.MappedSuperclass;
import java.time.Instant;

@MappedSuperclass
public abstract class AIContentEntity {

    @Column(name = "ai_generated", nullable = false)
    private boolean aiGenerated = false;

    @Column(name = "ai_model", length = 100)
    private String aiModel = "";

    @Column(name = "ai_system", length = 200)
    private String aiSystem = "";

    @Column(name = "ai_timestamp")
    private Instant aiTimestamp;

    @Column(name = "ai_human_edited", nullable = false)
    private boolean aiHumanEdited = false;

    // Getters — no setters, use a builder or constructor for immutability
    public boolean isAiGenerated() { return aiGenerated; }
    public String getAiModel() { return aiModel; }
    public String getAiSystem() { return aiSystem; }
    public Instant getAiTimestamp() { return aiTimestamp; }
    public boolean isAiHumanEdited() { return aiHumanEdited; }

    public String getAiDisclosureText() {
        if (!aiGenerated) return "";
        String edited = aiHumanEdited ? " (human-edited)" : "";
        return "Generated by AI system: %s, model: %s%s".formatted(aiSystem, aiModel, edited);
    }

    // Protected constructor for JPA and subclasses
    protected AIContentEntity() {}

    protected AIContentEntity(boolean aiGenerated, String aiModel, String aiSystem,
                              Instant aiTimestamp, boolean aiHumanEdited) {
        this.aiGenerated = aiGenerated;
        this.aiModel = aiModel;
        this.aiSystem = aiSystem;
        this.aiTimestamp = aiTimestamp;
        this.aiHumanEdited = aiHumanEdited;
    }
}
```

## AOP Aspect for Compliance Logging

```java
// ComplianceLoggingAspect.java
package com.example.compliance;

import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.Around;
import org.aspectj.lang.annotation.Aspect;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.stereotype.Component;
import java.time.Instant;
import java.util.Arrays;

@Aspect
@Component
public class ComplianceLoggingAspect {

    private static final Logger log = LoggerFactory.getLogger("ai.compliance.audit");

    private final AIComplianceProperties props;

    public ComplianceLoggingAspect(AIComplianceProperties props) {
        this.props = props;
    }

    @Around("@annotation(AuditAIInteraction)")
    public Object auditAiInteraction(ProceedingJoinPoint joinPoint) throws Throwable {
        if (!props.audit().enabled()) {
            return joinPoint.proceed();
        }

        String method = joinPoint.getSignature().toShortString();
        String args = Arrays.toString(joinPoint.getArgs());
        Instant start = Instant.now();

        Object result = joinPoint.proceed();

        log.info("ai_interaction method={} args={} provider={} model={} system={} "
                 + "timestamp={} duration_ms={}",
            method, truncate(args, 200), props.provider(), props.model(),
            props.systemName(), start, Instant.now().toEpochMilli() - start.toEpochMilli()
        );

        return result;
    }

    private static String truncate(String s, int max) {
        return s.length() <= max ? s : s.substring(0, max) + "...";
    }
}
```

## REST Controller Advice

```java
// AIMetadataAdvice.java
package com.example.compliance;

import org.springframework.core.MethodParameter;
import org.springframework.http.MediaType;
import org.springframework.http.converter.HttpMessageConverter;
import org.springframework.http.server.ServerHttpRequest;
import org.springframework.http.server.ServerHttpResponse;
import org.springframework.web.bind.annotation.RestControllerAdvice;
import org.springframework.web.servlet.mvc.method.annotation.ResponseBodyAdvice;
import java.time.Instant;
import java.util.Map;

@RestControllerAdvice(basePackages = "com.example.ai")
public class AIMetadataAdvice implements ResponseBodyAdvice<Object> {

    private final AIComplianceProperties props;

    public AIMetadataAdvice(AIComplianceProperties props) {
        this.props = props;
    }

    @Override
    public boolean supports(MethodParameter returnType,
                            Class<? extends HttpMessageConverter<?>> converterType) {
        return props.enabled();
    }

    @Override
    public Object beforeBodyWrite(Object body, MethodParameter returnType,
                                  MediaType mediaType,
                                  Class<? extends HttpMessageConverter<?>> converterType,
                                  ServerHttpRequest request, ServerHttpResponse response) {
        Map<String, Object> compliance = Map.of(
            "ai_generated", true,
            "ai_provider", props.provider(),
            "ai_model", props.model(),
            "ai_system", props.systemName(),
            "timestamp", Instant.now().toString(),
            "disclosure", "Content generated by AI per EU AI Act Art. 50"
        );
        return Map.of("data", body, "compliance", compliance);
    }
}
```
