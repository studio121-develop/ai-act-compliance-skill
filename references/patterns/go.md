# Go — Compliance at the Speed of... Go

Go's standard library already has everything you need for HTTP middleware — no npm install,
no pip install, just `import "net/http"`. Below are patterns that work with the standard
library, Chi, and Gin. Pick your favorite; compliance doesn't care about your router choice.

## Struct Definitions

```go
// compliance.go
package compliance

import "time"

// AIMetadata is attached to every AI-generated response.
type AIMetadata struct {
	AIGenerated  bool      `json:"ai_generated"`
	AIProvider   string    `json:"ai_provider"`
	AIModel      string    `json:"ai_model"`
	AISystem     string    `json:"ai_system"`
	Timestamp    time.Time `json:"timestamp"`
	Disclosure   string    `json:"disclosure"`
	HumanEdited  bool      `json:"human_edited"`
}

// ComplianceLog records an AI interaction for audit purposes.
type ComplianceLog struct {
	Endpoint        string     `json:"endpoint"`
	Method          string     `json:"method"`
	UserID          string     `json:"user_id,omitempty"`
	RequestSummary  string     `json:"request_summary"`
	ResponseSummary string     `json:"response_summary"`
	Metadata        AIMetadata `json:"metadata"`
	LoggedAt        time.Time  `json:"logged_at"`
}

// AIResponse wraps any payload with compliance metadata.
type AIResponse[T any] struct {
	Data       T          `json:"data"`
	Compliance AIMetadata `json:"compliance"`
}

// NewMetadata creates metadata with sensible defaults.
func NewMetadata(provider, model, system string) AIMetadata {
	return AIMetadata{
		AIGenerated: true,
		AIProvider:  provider,
		AIModel:     model,
		AISystem:    system,
		Timestamp:   time.Now().UTC(),
		Disclosure:  "Content generated by AI per EU AI Act Art. 50",
	}
}
```

## HTTP Middleware (net/http compatible)

```go
// middleware.go
package compliance

import (
	"net/http"
	"strings"
	"time"
)

var aiRoutePrefixes = []string{"/api/ai/", "/api/chat/", "/api/generate/"}

// Config holds the middleware configuration.
type Config struct {
	AIProvider string
	AISystem   string
}

// Middleware adds AI Act compliance headers to AI-related routes.
// Works with net/http, Chi, and anything that uses http.Handler.
func Middleware(cfg Config) func(http.Handler) http.Handler {
	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			if isAIRoute(r.URL.Path) {
				w.Header().Set("X-AI-Generated", "true")
				w.Header().Set("X-AI-Provider", cfg.AIProvider)
				w.Header().Set("X-AI-System", cfg.AISystem)
				w.Header().Set("X-AI-Act-Disclosure", "Article 50(1)")
				w.Header().Set("X-AI-Timestamp", time.Now().UTC().Format(time.RFC3339))
			}
			next.ServeHTTP(w, r)
		})
	}
}

func isAIRoute(path string) bool {
	for _, prefix := range aiRoutePrefixes {
		if strings.HasPrefix(path, prefix) {
			return true
		}
	}
	return false
}
```

## Gin Middleware Variant

```go
// gin_middleware.go
package compliance

import (
	"time"
	"github.com/gin-gonic/gin"
)

// GinMiddleware returns a Gin-compatible middleware.
func GinMiddleware(cfg Config) gin.HandlerFunc {
	return func(c *gin.Context) {
		if isAIRoute(c.Request.URL.Path) {
			c.Header("X-AI-Generated", "true")
			c.Header("X-AI-Provider", cfg.AIProvider)
			c.Header("X-AI-System", cfg.AISystem)
			c.Header("X-AI-Act-Disclosure", "Article 50(1)")
			c.Header("X-AI-Timestamp", time.Now().UTC().Format(time.RFC3339))
		}
		c.Next()
	}
}
```

## Structured Logging with slog

```go
// logger.go
package compliance

import (
	"context"
	"log/slog"
	"time"
)

// AuditLogger writes compliance audit logs using slog.
type AuditLogger struct {
	logger *slog.Logger
}

// NewAuditLogger creates a logger for compliance events.
func NewAuditLogger(logger *slog.Logger) *AuditLogger {
	return &AuditLogger{logger: logger.With("component", "ai_compliance")}
}

// LogInteraction records an AI interaction for audit trail.
func (a *AuditLogger) LogInteraction(ctx context.Context, log ComplianceLog) {
	a.logger.LogAttrs(ctx, slog.LevelInfo, "ai_interaction",
		slog.String("endpoint", log.Endpoint),
		slog.String("method", log.Method),
		slog.String("user_id", log.UserID),
		slog.String("request_summary", log.RequestSummary),
		slog.String("response_summary", log.ResponseSummary),
		slog.Bool("ai_generated", log.Metadata.AIGenerated),
		slog.String("ai_provider", log.Metadata.AIProvider),
		slog.String("ai_model", log.Metadata.AIModel),
		slog.String("ai_system", log.Metadata.AISystem),
		slog.Time("ai_timestamp", log.Metadata.Timestamp),
		slog.Time("logged_at", time.Now().UTC()),
	)
}
```

## Full Handler Example

```go
// handler.go
package main

import (
	"encoding/json"
	"log/slog"
	"net/http"
	"os"

	"yourapp/compliance"
)

func main() {
	cfg := compliance.Config{
		AIProvider: os.Getenv("AI_PROVIDER"),
		AISystem:   os.Getenv("AI_SYSTEM_NAME"),
	}
	auditLogger := compliance.NewAuditLogger(slog.Default())

	mux := http.NewServeMux()
	mux.HandleFunc("POST /api/ai/chat", chatHandler(cfg, auditLogger))

	handler := compliance.Middleware(cfg)(mux)

	slog.Info("starting server", "addr", ":8080")
	http.ListenAndServe(":8080", handler)
}

func chatHandler(cfg compliance.Config, audit *compliance.AuditLogger) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		var req struct {
			Message string `json:"message"`
		}
		if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
			http.Error(w, `{"error":"invalid request body"}`, http.StatusBadRequest)
			return
		}

		aiContent := "AI response to: " + req.Message
		meta := compliance.NewMetadata(cfg.AIProvider, os.Getenv("AI_MODEL"), cfg.AISystem)

		go audit.LogInteraction(r.Context(), compliance.ComplianceLog{
			Endpoint:        r.URL.Path,
			Method:          r.Method,
			RequestSummary:  req.Message,
			ResponseSummary: aiContent,
			Metadata:        meta,
		})

		resp := compliance.AIResponse[string]{Data: aiContent, Compliance: meta}
		w.Header().Set("Content-Type", "application/json")
		json.NewEncoder(w).Encode(resp)
	}
}
```
