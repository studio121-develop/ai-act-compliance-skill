# FastAPI — Because Your AI API Should Also Be Compliant

FastAPI already generates your docs, validates your inputs, and makes you look good in code
reviews. Now let's make it handle AI Act compliance too. Pydantic models for metadata,
middleware for headers, dependency injection for logging — it's almost like this framework
was designed for regulation. (It wasn't, but we'll pretend.)

## Compliance Middleware

```python
# middleware.py
import time
from datetime import datetime, timezone
from starlette.middleware.base import BaseHTTPMiddleware
from starlette.requests import Request
from starlette.responses import Response

AI_ROUTE_PREFIXES = ("/api/ai/", "/api/chat/", "/api/generate/")


class AIComplianceMiddleware(BaseHTTPMiddleware):
    def __init__(self, app, ai_provider: str = "unknown", ai_system: str = "unknown"):
        super().__init__(app)
        self.ai_provider = ai_provider
        self.ai_system = ai_system

    async def dispatch(self, request: Request, call_next) -> Response:
        start_time = time.monotonic()
        response = await call_next(request)

        if any(request.url.path.startswith(prefix) for prefix in AI_ROUTE_PREFIXES):
            response.headers["X-AI-Generated"] = "true"
            response.headers["X-AI-Provider"] = self.ai_provider
            response.headers["X-AI-System"] = self.ai_system
            response.headers["X-AI-Act-Disclosure"] = "Article 50(1)"
            response.headers["X-AI-Timestamp"] = datetime.now(timezone.utc).isoformat()
            elapsed = time.monotonic() - start_time
            response.headers["X-AI-Processing-Time-Ms"] = str(int(elapsed * 1000))

        return response
```

## Pydantic Models

```python
# models.py
from datetime import datetime, timezone
from pydantic import BaseModel, Field
from enum import Enum


class AIRiskLevel(str, Enum):
    MINIMAL = "minimal"
    LIMITED = "limited"
    HIGH = "high"
    UNACCEPTABLE = "unacceptable"


class AIContentMetadata(BaseModel):
    ai_generated: bool = True
    ai_provider: str
    ai_model: str
    ai_system: str
    timestamp: datetime = Field(default_factory=lambda: datetime.now(timezone.utc))
    risk_level: AIRiskLevel = AIRiskLevel.LIMITED
    disclosure: str = "Content generated by AI per EU AI Act Art. 50"


class AIDisclosure(BaseModel):
    system_name: str
    provider_name: str
    purpose: str
    risk_level: AIRiskLevel
    human_oversight: bool
    explanation: str = Field(
        default="You are interacting with an AI system. "
        "This disclosure is required by EU AI Act Article 50."
    )


class ComplianceAuditLog(BaseModel):
    endpoint: str
    method: str
    user_id: str | None = None
    request_summary: str
    response_summary: str
    metadata: AIContentMetadata
    logged_at: datetime = Field(default_factory=lambda: datetime.now(timezone.utc))
    retention_days: int = 365


class AIResponse(BaseModel):
    """Wraps any AI response with compliance metadata."""
    content: str
    compliance: AIContentMetadata
```

## Dependency Injection for Compliance Logging

```python
# dependencies.py
import logging
from datetime import datetime, timezone
from fastapi import Request, BackgroundTasks
from models import ComplianceAuditLog, AIContentMetadata

logger = logging.getLogger("ai_compliance")


class ComplianceLogger:
    def __init__(self, request: Request, background_tasks: BackgroundTasks):
        self.request = request
        self.background_tasks = background_tasks

    def log(
        self,
        request_summary: str,
        response_summary: str,
        metadata: AIContentMetadata,
        user_id: str | None = None,
    ) -> None:
        audit_log = ComplianceAuditLog(
            endpoint=str(self.request.url.path),
            method=self.request.method,
            user_id=user_id,
            request_summary=request_summary,
            response_summary=response_summary,
            metadata=metadata,
        )
        # Run in background so we don't block the response
        self.background_tasks.add_task(_persist_audit_log, audit_log)


async def _persist_audit_log(log: ComplianceAuditLog) -> None:
    """Replace with your actual persistence layer."""
    logger.info("compliance_audit", extra=log.model_dump(mode="json"))
```

## Full Compliant Router

```python
# router.py
import os
from fastapi import APIRouter, Depends, HTTPException, BackgroundTasks, Request
from pydantic import BaseModel
from models import AIContentMetadata, AIDisclosure, AIRiskLevel, AIResponse
from dependencies import ComplianceLogger

router = APIRouter(prefix="/api/ai", tags=["ai"])

AI_PROVIDER = os.getenv("AI_PROVIDER", "unknown")
AI_MODEL = os.getenv("AI_MODEL", "unknown")
AI_SYSTEM = os.getenv("AI_SYSTEM_NAME", "unknown")


class ChatRequest(BaseModel):
    message: str
    conversation_id: str | None = None


@router.get("/disclosure")
async def get_disclosure() -> AIDisclosure:
    return AIDisclosure(
        system_name=AI_SYSTEM,
        provider_name=AI_PROVIDER,
        purpose="General-purpose conversational AI assistant",
        risk_level=AIRiskLevel.LIMITED,
        human_oversight=True,
    )


@router.post("/chat")
async def chat(
    body: ChatRequest,
    request: Request,
    background_tasks: BackgroundTasks,
) -> AIResponse:
    compliance_logger = ComplianceLogger(request, background_tasks)
    metadata = AIContentMetadata(
        ai_provider=AI_PROVIDER,
        ai_model=AI_MODEL,
        ai_system=AI_SYSTEM,
    )

    # Replace with your actual AI call
    ai_content = f"AI response to: {body.message}"

    compliance_logger.log(
        request_summary=body.message[:200],
        response_summary=ai_content[:200],
        metadata=metadata,
    )

    return AIResponse(content=ai_content, compliance=metadata)
```

## App Setup

```python
# main.py
import os
from fastapi import FastAPI
from middleware import AIComplianceMiddleware
from router import router as ai_router

app = FastAPI(title="AI-Compliant API", version="1.0.0")

app.add_middleware(
    AIComplianceMiddleware,
    ai_provider=os.getenv("AI_PROVIDER", "unknown"),
    ai_system=os.getenv("AI_SYSTEM_NAME", "unknown"),
)

app.include_router(ai_router)
```
